171 void run_command(uint32_t cmd)
172 {
173  int8_t error = 0;
174  uint32_t conv_time = 0;
175  uint32_t user_command;
176  int8_t readIC=0;
177  char input = 0;
178  switch (cmd)
179  {
180 
181  case 1: // Write Configuration Register
182  wakeup_sleep(TOTAL_IC);
183  LTC6811_wrcfg(TOTAL_IC,bms_ic);
184  print_config();
185  break;
186 
187  case 2: // Read Configuration Register
188  wakeup_sleep(TOTAL_IC);
189  error = LTC6811_rdcfg(TOTAL_IC,bms_ic);
190  check_error(error);
191  print_rxconfig();
192  break;
193 
194  case 3: // Start Cell ADC Measurement
195  wakeup_sleep(TOTAL_IC);
196  LTC6811_adcv(ADC_CONVERSION_MODE,ADC_DCP,CELL_CH_TO_CONVERT);
197  conv_time = LTC6811_pollAdc();
198  Serial.print(F("cell conversion completed in:"));
199  Serial.print(((float)conv_time/1000), 1);
200  Serial.println(F("mS"));
201  Serial.println();
202  break;
203 
204  case 4: // Read Cell Voltage Registers
205  wakeup_sleep(TOTAL_IC);
206  error = LTC6811_rdcv(0, TOTAL_IC,bms_ic); // Set to read back all cell voltage registers
207  check_error(error);
208  print_cells(DATALOG_DISABLED);
209  break;
210 
211  case 5: // Start GPIO ADC Measurement
212  wakeup_sleep(TOTAL_IC);
213  LTC6811_adax(ADC_CONVERSION_MODE , AUX_CH_TO_CONVERT);
214  LTC6811_pollAdc();
215  Serial.println(F("aux conversion completed"));
216  Serial.println();
217  break;
218 
219  case 6: // Read AUX Voltage Registers
220  wakeup_sleep(TOTAL_IC);
221  error = LTC6811_rdaux(0,TOTAL_IC,bms_ic); // Set to read back all aux registers
222  check_error(error);
223  print_aux(DATALOG_DISABLED);
224  break;
225 
226  case 7: // Start Status ADC Measurement
227  wakeup_sleep(TOTAL_IC);
228  LTC6811_adstat(ADC_CONVERSION_MODE, STAT_CH_TO_CONVERT);
229  LTC6811_pollAdc();
230  Serial.println(F("stat conversion completed"));
231  Serial.println();
232  break;
233 
234  case 8: // Read Status registers
235  wakeup_sleep(TOTAL_IC);
236  error = LTC6811_rdstat(0,TOTAL_IC,bms_ic); // Set to read back all aux registers
237  check_error(error);
238  print_stat();
239  break;
240 
241  case 9: // Loop Measurements
242  Serial.println(F("transmit 'm' to quit"));
243  wakeup_sleep(TOTAL_IC);
244  LTC6811_wrcfg(TOTAL_IC,bms_ic);
245  while (input != 'm')
246  {
247  if (Serial.available() > 0)
248  {
249  input = read_char();
250  }
251 
252  measurement_loop(DATALOG_DISABLED);
253 
254  delay(MEASUREMENT_LOOP_TIME);
255  }
256  //print_menu();
257  break;
258 
259  case 10: // Run open wire self test
260  print_pec();
261 
262  break;
263 
264  case 11: // Read in raw configuration data
265  LTC6811_reset_crc_count(TOTAL_IC,bms_ic);
266  break;
267 
268  case 12: // Run the ADC/Memory Self Test
269  wakeup_sleep(TOTAL_IC);
270  error = LTC6811_run_cell_adc_st(CELL,ADC_CONVERSION_MODE,bms_ic);
271  Serial.print(error, DEC);
272  Serial.println(F(" : errors detected in Digital Filter and CELL Memory \n"));
273 
274  wakeup_sleep(TOTAL_IC);
275  error = LTC6811_run_cell_adc_st(AUX,ADC_CONVERSION_MODE, bms_ic);
276  Serial.print(error, DEC);
277  Serial.println(F(" : errors detected in Digital Filter and AUX Memory \n"));
278 
279  wakeup_sleep(TOTAL_IC);
280  error = LTC6811_run_cell_adc_st(STAT,ADC_CONVERSION_MODE, bms_ic);
281  Serial.print(error, DEC);
282  Serial.println(F(" : errors detected in Digital Filter and STAT Memory \n"));
283  print_menu();
284  break;
285 
286  case 13: // Enable a discharge transistor
287  Serial.println(F("Please enter the Spin number"));
288  readIC = (int8_t)read_int();
289  LTC6811_set_discharge(readIC,TOTAL_IC,bms_ic);
290  wakeup_sleep(TOTAL_IC);
291  LTC6811_wrcfg(TOTAL_IC,bms_ic);
292  print_config();
293  break;
294 
295  case 14: // Clear all discharge transistors
296  clear_discharge(TOTAL_IC,bms_ic);
297  wakeup_sleep(TOTAL_IC);
298  LTC6811_wrcfg(TOTAL_IC,bms_ic);
299  print_config();
300  break;
301 
302  case 15: // Clear all ADC measurement registers
303  wakeup_sleep(TOTAL_IC);
304  LTC6811_clrcell();
305  LTC6811_clraux();
306  LTC6811_clrstat();
307  Serial.println(F("All Registers Cleared"));
308  break;
309 
310  case 16: // Run the Mux Decoder Self Test
311  wakeup_sleep(TOTAL_IC);
312  LTC6811_diagn();
313  delay(5);
314  error = LTC6811_rdstat(0,TOTAL_IC,bms_ic); // Set to read back all aux registers
315  check_error(error);
316  error = 0;
317  for (int ic = 0; ic<TOTAL_IC; ic++)
318  {
319  if (bms_ic[ic].stat.mux_fail[0] != 0) error++;
320  }
321  if (error==0) Serial.println(F("Mux Test: PASS "));
322  else Serial.println(F("Mux Test: FAIL "));
323 
324  break;
325 
326  case 17: // Run ADC Overlap self test
327  wakeup_sleep(TOTAL_IC);
328  error = (int8_t)LTC6811_run_adc_overlap(TOTAL_IC,bms_ic);
329  if (error==0) Serial.println(F("Overlap Test: PASS "));
330  else Serial.println(F("Overlap Test: FAIL"));
331  break;
332 
333  case 18: // Run ADC Redundancy self test
334  wakeup_sleep(TOTAL_IC);
335  error = LTC6811_run_adc_redundancy_st(ADC_CONVERSION_MODE,AUX,TOTAL_IC, bms_ic);
336  Serial.print(error, DEC);
337  Serial.println(F(" : errors detected in AUX Measurement \n"));
338 
339  wakeup_sleep(TOTAL_IC);
340  error = LTC6811_run_adc_redundancy_st(ADC_CONVERSION_MODE,STAT,TOTAL_IC, bms_ic);
341  Serial.print(error, DEC);
342  Serial.println(F(" : errors detected in STAT Measurement \n"));
343  break;
344 
345  case 19:
346  LTC6811_run_openwire(TOTAL_IC, bms_ic);
347  print_open();
348  break;
349 
350  case 20: //Datalog print option Loop Measurements
351  Serial.println(F("transmit 'm' to quit"));
352  wakeup_sleep(TOTAL_IC);
353  LTC6811_wrcfg(TOTAL_IC,bms_ic);
354  while (input != 'm')
355  {
356  if (Serial.available() > 0)
357  {
358  input = read_char();
359  }
360 
361  measurement_loop(DATALOG_ENABLED);
362 
363  delay(MEASUREMENT_LOOP_TIME);
364  }
365  print_menu();
366  break;
367 
368  case 'm': //prints menu
369  print_menu();
370  break;
371 
372  default:
373  Serial.println(F("Incorrect Option"));
374  break;
375  }
376 }

378 void measurement_loop(uint8_t datalog_en)
379 {
380  int8_t error = 0;
381  if (WRITE_CONFIG == ENABLED)
382  {
383  wakeup_sleep(TOTAL_IC);
384  LTC6811_wrcfg(TOTAL_IC,bms_ic);
385  print_config();
386  }
387 
388  if (READ_CONFIG == ENABLED)
389  {
390  wakeup_sleep(TOTAL_IC);
391  error = LTC6811_rdcfg(TOTAL_IC,bms_ic);
392  check_error(error);
393  print_rxconfig();
394  }
395 
396  if (MEASURE_CELL == ENABLED)
397  {
398  wakeup_idle(TOTAL_IC);
399  LTC6811_adcv(ADC_CONVERSION_MODE,ADC_DCP,CELL_CH_TO_CONVERT);
400  LTC6811_pollAdc();
401  wakeup_idle(TOTAL_IC);
402  error = LTC6811_rdcv(0, TOTAL_IC,bms_ic);
403  check_error(error);
404  print_cells(datalog_en);
405 
406  }
407 
408  if (MEASURE_AUX == ENABLED)
409  {
410  wakeup_idle(TOTAL_IC);
411  LTC6811_adax(ADC_CONVERSION_MODE , AUX_CH_ALL);
412  LTC6811_pollAdc();
413  wakeup_idle(TOTAL_IC);
414  error = LTC6811_rdaux(0,TOTAL_IC,bms_ic); // Set to read back all aux registers
415  check_error(error);
416  print_aux(datalog_en);
417  }
418 
419  if (MEASURE_STAT == ENABLED)
420  {
421  wakeup_idle(TOTAL_IC);
422  LTC6811_adstat(ADC_CONVERSION_MODE, STAT_CH_ALL);
423  LTC6811_pollAdc();
424  wakeup_idle(TOTAL_IC);
425  error = LTC6811_rdstat(0,TOTAL_IC,bms_ic); // Set to read back all aux registers
426  check_error(error);
427  print_stat();
428  }
429 
430  if (PRINT_PEC == ENABLED)
431  {
432  print_pec();
433  }
434 
435 }
